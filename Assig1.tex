{\LARGE  \textbf{Problem 2 (a)}}\\
{\Large \textbf{Evaluate without using  trigonometric tables,}\\
{\Large \textbf{$\sin^2 28^{\circ}+\sin^2 62^{\circ}+\tan^{2}38^{\circ}-\cot^{2}52^{\circ}+\sec^{2}30^{\circ}$\\}}\\
{\LARGE \textbf{Solution:}}\\
(a)  
\begin{align}
x &=\sin^{2}28^{\circ} + \sin^{2}62^{\circ} + \tan^{2}38^{\circ}-\cot^{2}52^{\circ}+\sec^{2}30^{\circ}\\
x &=\sin^{2}28^{\circ} + \sin^{2}(90^{\circ} - 28^{\circ}) + tan^{2}38^{\circ} - cot^{2}(90^{\circ} - 38^{\circ}) + (\frac{2}{\sqrt{3}})^{2}\\ 
x &=\sin^{2}28^{\circ} + \cos^{2}28^{\circ} + \tan^{2}38^{\circ} - \tan^{2}38^{\circ} + (\frac{4}{3})\\
x &= 1 + \frac{4}{3} = \frac{3+4}{3} = \frac{7}{3} = 2\frac{1}{3}\\
\therefore x &= \frac{7}{3}\\
\end{align}
